async function fetchAllCoins(client: SuiClient, address: string): Promise<Coin[]> {

    setStatus(null);

      { transaction: tx },

        onSuccess: async (res) => {

          const msg = err instanceof Error ? err.message : String(err);
          setStatus(`Error: ${msg}`);
          toast.error(`Merge failed: ${msg}`);
        },
